CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

# Leaf Directories (does not contain a CMakeLists.txt)
SET(LEAF_CHILDREN
	"Platform"
	"Misc"
	"Class"
	"Templates"
	"Containers"
	)

# Similar to Leaf Directories but not included in the list of include directories.
SET(HIDDEN_LEAF_CHILDREN
	"PlatformCommon"
	)

IF(APPLE)
	SET(HIDDEN_LEAF_CHILDREN ${HIDDEN_LEAF_CHILDREN} "PlatformMac")
ENDIF(APPLE)

#####################################################################
############## SHOULD NOT NEED TO EDIT BELOW THIS LINE ##############
#####################################################################

SET(TCC_ENGINE_CORE_INCLUDE_DIRS)
SET(TCC_ENGINE_CORE_INCLUDES)
SET(TCC_ENGINE_CORE_SOURCES)

FOREACH(CHILD ${LEAF_CHILDREN})
	SET(TCC_ENGINE_CORE_INCLUDE_DIRS ${TCC_ENGINE_CORE_INCLUDE_DIRS} "${CMAKE_CURRENT_SOURCE_DIR}/${CHILD}")

	FILE(GLOB CHILD_INCLUDES "${CHILD}/*.h" "${CHILD}/*.hpp")
	SET(TCC_ENGINE_CORE_INCLUDES ${TCC_ENGINE_CORE_INCLUDES} ${CHILD_INCLUDES})

	FILE(GLOB CHILD_SOURCES "${CHILD}/*.cpp" "${CHILD}/*.mm")
	SET(TCC_ENGINE_CORE_SOURCES ${TCC_ENGINE_CORE_SOURCES} ${CHILD_SOURCES})
ENDFOREACH(CHILD)

FOREACH(CHILD ${HIDDEN_LEAF_CHILDREN})
	FILE(GLOB CHILD_INCLUDES "${CHILD}/*.h" "${CHILD}/*.hpp")
	SET(TCC_ENGINE_CORE_INCLUDES ${TCC_ENGINE_CORE_INCLUDES} ${CHILD_INCLUDES})

	FILE(GLOB CHILD_SOURCES "${CHILD}/*.cpp" "${CHILD}/*.mm")
	SET(TCC_ENGINE_CORE_SOURCES ${TCC_ENGINE_CORE_SOURCES} ${CHILD_SOURCES})
ENDFOREACH(CHILD)

# Exported to Parent Scope
SET(TCC_ENGINE_CORE_INCLUDE_DIRS 
	${TCC_ENGINE_CORE_INCLUDE_DIRS} 
	${CMAKE_CURRENT_SOURCE_DIR}
	PARENT_SCOPE
	)
SET(TCC_ENGINE_CORE_INCLUDES 
	${TCC_ENGINE_CORE_INCLUDES} 
	"${CMAKE_CURRENT_SOURCE_DIR}/Core.h"
	PARENT_SCOPE
	)
SET(TCC_ENGINE_CORE_SOURCES 
	${TCC_ENGINE_CORE_SOURCES} 
	PARENT_SCOPE
	)
